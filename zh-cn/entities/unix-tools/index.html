# 分卷压缩

# tar
tar cfz bigfile.tgz bigfile
# zip 
zip ./bigfile.zip bigfile
# gzip 就地压缩，原文件被压缩文件替换
gzip bigfile
# bzip2 类似 gzip
bzip bigfile
# xz 类似 gzip 花费更多时间，压缩率更高
xz bigfile
